Basic Demo

Typed.js^500 is a JavaScript library. It types out sentences. `And` `then` `deletes` `them`. Try it out!
Typed 
    
  var typed = new Typed("#typed", {
    stringsElement: '#typed-strings',
    typeSpeed: 0,
    backSpeed: 0,
    backDelay: 500,
    startDelay: 1000,
    loop: false,
    onBegin: function(self) { prettyLog('onBegin ' + self) },
    onComplete: function(self) { prettyLog('onCmplete ' + self) },
    preStringTyped: function(pos, self) { prettyLog('preStringTyped ' + pos + ' ' + self); },
    onStringTyped: function(pos, self) { prettyLog('onStringTyped ' + pos + ' ' + self) },
    onLastStringBackspaced: function(self) { prettyLog('onLastStringBackspaced ' + self) },
    onTypingPaused: function(pos, self) { prettyLog('onTypingPaused ' + pos + ' ' + self) },
    onTypingResumed: function(pos, self) { prettyLog('onTypingResumed ' + pos + ' ' + self) },
    onReset: function(self) { prettyLog('onReset ' + self) },
    onStop: function(pos, self) { prettyLog('onStop ' + pos + ' ' + self) },
    onStart: function(pos, self) { prettyLog('onStart ' + pos + ' ' + self) },
    onDestroy: function(self) { prettyLog('onDestroy ' + self) }
  });

Next String Delay

    
  document.addEventListener('DOMContentLoaded', function() {
    console.log("Initializing Typed.js demo for nextStringDelay...");
    try {
      var typed7 = new Typed('#typed7', {
        strings: ['First string', 'Second string', 'Third string'],
        typeSpeed: 40,
        backSpeed: 20,
        backDelay: 1000,
        nextStringDelay: 1500, // 1.5s pause before typing the next string
        loop: true
      });

      // Control buttons for the Next String Delay demo
      document.querySelector('.toggle7').addEventListener('click', function() { typed7.toggle(); });
      document.querySelector('.start7').addEventListener('click', function() { typed7.start(); });
      document.querySelector('.stop7').addEventListener('click', function() { typed7.stop(); });
      document.querySelector('.reset7').addEventListener('click', function() { typed7.reset(); });
      document.querySelector('.destroy7').addEventListener('click', function() { typed7.destroy(); });
      document.querySelector('.loop7').addEventListener('click', function() { typed7.loop = !typed7.loop; });
    } catch (error) {
      console.error('Error initializing Typed.js demo:', error);
    }
  });
